Basic Usage

// Initialize harmonic analysis modules const musicTheory = new MusicTheoryUtils(); const scaleManager = new ScaleKeyManager(); const harmonicAnalysis = new HarmonicAnalysis(); const chordEngine = new ChordProgressionEngine(scaleManager); // Set key and mode scaleManager.setKey('C', 'major'); const scale = scaleManager.getScale('C', 'major'); console.log('C Major scale:', scale); // ['C', 'D', 'E', 'F', 'G', 'A', 'B'] // Generate chord progression const progression = chordEngine.generateProgression('C', 'major', 'common', 4); console.log('Chord progression:', progression); // Analyze chord const chord = musicTheory.analyzeChord(['C4', 'E4', 'G4']); console.log('Chord analysis:', chord); // { root: 'C4', quality: 'major', ... }

Pattern Integration

// Integrate with pattern generation const patternIntegration = new PatternHarmonicIntegration(app); // Set harmonic context patternIntegration.setHarmonicContext('G', 'minor'); // Generate harmonically constrained patterns const bassPattern = patternIntegration.generateHarmonicPatterns('bass', 16); const leadPattern = patternIntegration.generateHarmonicPatterns('lead', 16); // Apply harmonic constraints to existing patterns const constrainedPatterns = patternIntegration.applyHarmonicConstraints(existingPatterns);
